Logs:
LOAD
    log_id,
    project_id,
    action_type,
    Timestamp#(timestamp, 'YYYY-MM-DD HH:MM:SS') AS timestamp
FROM [http://localhost:3000/project-action-logs]
(json);


LatestLogs:
LOAD 
    project_id,
    Timestamp(Max(timestamp)) AS latest_timestamp
RESIDENT Logs
GROUP BY project_id;


JOIN (Logs)
LOAD project_id,
     latest_timestamp
RESIDENT LatestLogs;

LatestLogsFiltered:
LOAD project_id,
     action_type as Current_Action,
     timestamp
RESIDENT Logs
WHERE timestamp = latest_timestamp;
